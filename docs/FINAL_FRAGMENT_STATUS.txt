📋 NAMING CONVENTION
════════════════════════════════════════════════════════════════════════════

Rule: All names must be identical

export const FRAGMENT_NAME = gql`
  fragment FRAGMENT_NAME on Type {    ← Fragment definition name
    field
  }
`;

// Usage in query:
{
  ...FRAGMENT_NAME                    ← Fragment spread name
}

All three must match!
